async function generateBatch() { const key = document.getElementById('apiKey').value.trim(); if(!key) return alert("Please paste your API Key first!"); const loader = document.getElementById('loader'); loader.style.display = "block"; loader.innerText = "Connecting to Gemini 3 AI..."; // 2026 STABLE ENDPOINT AND MODEL const apiURL = `https://generativelanguage.googleapis.com/v1/models/gemini-3-flash:generateContent?key=${key}`; const prompt = "Generate 5 Islamic Quiz questions. Return ONLY a valid JSON array. Format: [{\"en\":{\"q\":\"Question?\",\"a\":\"A\",\"options\":{\"A\":\"Ans1\",\"B\":\"Ans2\",\"C\":\"Ans3\",\"D\":\"Ans4\"},\"exp\":\"Explanation\"},\"ur\":{\"q\":\"سوال؟\",\"a\":\"A\",\"options\":{\"A\":\"جواب1\",\"B\":\"جواب2\",\"C\":\"جواب3\",\"D\":\"جواب4\"},\"exp\":\"وضاحت\"},\"hi\":{\"q\":\"Sawal?\",\"a\":\"A\",\"options\":{\"A\":\"Jawab1\",\"B\":\"Jawab2\",\"C\":\"Jawab3\",\"D\":\"Jawab4\"},\"exp\":\"Explanation\"}}]"; try { const response = await fetch(apiURL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ contents: [{ parts: [{ text: prompt }] }], generationConfig: { response_mime_type: "application/json" } }) }); // SAFETY CHECK: If response is not OK, stop before parsing JSON if (!response.ok) { const errorText = await response.text(); // Get raw HTML/text error console.error("Server Error Response:", errorText); throw new Error(`HTTP ${response.status}: The API URL or Model might be incorrect.`); } const result = await response.json(); const rawText = result.candidates[0].content.parts[0].text; currentBatch = JSON.parse(rawText); displayBatch(); loader.style.display = "none"; } catch (e) { console.error("DEBUG INFO:", e); loader.innerText = "❌ Error: " + e.message; alert("System Message: " + e.message); } }
